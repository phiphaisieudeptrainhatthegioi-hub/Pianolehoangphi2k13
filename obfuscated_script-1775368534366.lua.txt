local Fluent = loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()
repeat task.wait() until Fluent

local Window = Fluent:CreateWindow({
    Title = "Piano",
    SubTitle = "",
    TabWidth = 160,
    Size = UDim2.fromOffset(400, 300),
    Acrylic = false, 
    Theme = "Dark",
    MinimizeKey = Enum.KeyCode.End 
})

if Window.Frame then
    Window.Frame.Active = false
end

local Tabs = {
    Main = Window:AddTab({ Title = "Sheet", Icon = "" })
}

local ScreenGui = Instance.new("ScreenGui")
local ImageButton = Instance.new("ImageButton")
local UICorner = Instance.new("UICorner")
ScreenGui.Parent = game.CoreGui
ScreenGui.Name = "Toggle"
ScreenGui.ResetOnSpawn = false
ImageButton.Parent = ScreenGui
ImageButton.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
ImageButton.Position = UDim2.new(0.1, 0, 0.2, 0)
ImageButton.Size = UDim2.new(0, 50, 0, 50)
ImageButton.Image = "http://www.roblox.com/asset/?id=831902769514"

UICorner.CornerRadius = UDim.new(1, 0) 
UICorner.Parent = ImageButton
ImageButton.MouseButton1Click:Connect(function()
    Window:Minimize(not Window.Minimized)
end)

local Vim = game:GetService("VirtualInputManager")
local playCount = 0 -- Dùng ID phiên chạy để tránh xung đột

local function PlaySheet(sheetText, speed)
    playCount = playCount + 1
    local currentSession = playCount
    _G.CurrentPianoSession = currentSession
    
    task.spawn(function()
        local i = 1
        local delayTime = 60 / (speed * 2)
        
        while i <= #sheetText do
            -- Kiểm tra nếu có phiên chạy mới hoặc yêu cầu dừng
            if _G.CurrentPianoSession ~= currentSession then break end
            
            local char = sheetText:sub(i, i)
            if char == "[" then
                local chord = ""
                i = i + 1
                while i <= #sheetText and sheetText:sub(i, i) ~= "]" do
                    chord = chord .. sheetText:sub(i, i)
                    i = i + 1
                end
                for j = 1, #chord do
                    local key = chord:sub(j, j)
                    Vim:SendKeyEvent(true, key:upper(), false, game)
                    task.wait(0.01)
                    Vim:SendKeyEvent(false, key:upper(), false, game)
                end
            elseif char:match("[%w%d%p]") and char ~= " " then
                Vim:SendKeyEvent(true, char:upper(), false, game)
                task.wait(0.01)
                Vim:SendKeyEvent(false, char:upper(), false, game)
            end
            task.wait(delayTime)
            i = i + 1
        end
    end)
end

local InputSheet = Tabs.Main:AddInput("Input", {
    Title = "Dán Sheet",
    Default = "",
    Callback = function() end
})

local InputSpeed = Tabs.Main:AddSlider("Speed", {
    Title = "BPM",
    Default = 100,
    Min = 20,
    Max = 600,
    Rounding = 0,
    Callback = function() end
})

Tabs.Main:AddButton({
    Title = "Chạy",
    Callback = function()
        PlaySheet(InputSheet.Value, InputSpeed.Value)
    end
})

Tabs.Main:AddButton({
    Title = "Dừng",
    Callback = function()
        _G.CurrentPianoSession = nil -- Hủy phiên chạy hiện tại
    end
})
